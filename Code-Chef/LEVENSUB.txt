//main
int main(){
    int tc;
    si(tc);
    while(tc--){
        int n;
        si(n);
        vi a(n,0);
        ll rsum=0,lsum=0;
        for(int i=0;i<n;i++){
            cin>>a[i];
            rsum+=a[i];
        }
        int ans=0;
        if(rsum%2==0){
            cout<<n<<endl;
            continue;
        }
        for(int i=0;i<n;i++){
            lsum += a[i];
            rsum -= a[i];
            if(lsum%2==0)
                ans = max(ans,i+1);
            if(rsum%2==0)
                ans = max(ans,n-1-i);
        }
        cout<<ans<<endl;
    }
    return 0;
}
