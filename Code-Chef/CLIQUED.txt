#include <bits/stdc++.h>
using namespace std;

typedef long long 			ll;
typedef vector<ll> 		vi; 
typedef pair<ll,ll> 		ii; 
#define sz(a) 				int((a).size()) 
#define pb 					push_back 
#define all(c) 				(c).begin(),(c).end() 
#define tr(c,i) 			for(auto i = (c).begin(); i != (c).end(); i++) 
#define present(c,x) 		((c).find(x) != (c).end()) 
#define cpresent(c,x) 		(find(all(c),x) != (c).end())  

int main(){
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);

	int tc;
	cin>>tc;
	while(tc--){
		
		ll n,k,x,m,s;
		cin>>n>>k>>x>>m>>s;
		vector<ii> Adj[n+1];
		

	
		for(int i=0; i< m; i++){
			ll x,y,w;
			cin>>x>>y>>w;
			Adj[x].pb({y,w});
			Adj[y].pb({x,w});

		}
		
	    queue<int> Q;
	    Q.push(s);
	    vi dist(n+1,2e18);
	    dist[s]=0;
		if(s<=k){
			
			for(int i=1; i<=k ; i++){
				dist[i] = min(dist[i],dist[s]+x);
				if(sz(Adj[i]))
	    			Q.push(i);
			}
		}

	    bool fl =true;
	  
	    while(!(Q.empty())){

	    	int u = Q.front();Q.pop();
	    	
	    	for(int i=0; i< sz(Adj[u]); i++){
	    		ii X = Adj[u][i];

	    		ll v = X.first,w = X.second;

	    		if(dist[v] > dist[u]+w){
	    			dist[v] = dist[u]+w;
	    			Q.push(v);
	    		}	

	    		if(v<=k and fl){
	    			fl = false;
	    			for(int i=1; i<=k ; i++){
	    				dist[i] = min(dist[i],dist[v]+x);
	    				if(sz(Adj[i]))
	    					Q.push(i); 
	    			}
	    		}

	    		
	    	}
	    }
	   	

	    ll mi = 2e18;

	    for(int i=1; i<= k; i++){
	    	mi = min(mi,dist[i]);
	    }
	    for(int i=1; i<= k; i++){
	    	dist[i] = min(dist[i],mi+x);
	    }


	    for(int i=0; i< n; i++){
	    	cout << dist[i+1] << " " ;
	    }
	    cout<<endl;

	}
	return 0;
}
