int main(){

	int k;
	cin>>k;
	string q;
	cin>>q;
	if(sz(q) < k ){

	}
	else{
		set<char> st(all(q));
		if(sz(st) < k){
			cout<<"NO";
		}
		else{
			cout<<"YES\n";
			vi a(27,1);
			int i=0;
			while(k--){
				string s="";
				bool start = true;
				for(; i< sz(q); i++){
					if(a[q[i]-'a'] and start){
						s += q[i];
						a[q[i]-'a']=0;
						start = false;

					}
					else if(a[q[i]-'a']==1 and start == false){
						break;
					}
					else{
						s += q[i];
					}
